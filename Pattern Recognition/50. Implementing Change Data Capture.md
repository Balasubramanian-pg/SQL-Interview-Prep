**Pattern**: Tracking all modifications to data over time.

**Decomposition Strategy**:

1. Create audit tables to track changes
2. Implement triggers or log-based capture
3. Record operation type, timestamp, and user
4. Design queries to analyze change history

**Example**: "Track all changes to customer data with audit trail."

```SQL
-- Create audit table
CREATE TABLE customer_audit (
    audit_id SERIAL PRIMARY KEY,
    customer_id INT NOT NULL,
    field_name VARCHAR(50) NOT NULL,
    old_value TEXT,
    new_value TEXT,
    change_type VARCHAR(10) NOT NULL, -- INSERT, UPDATE, DELETE
    changed_by VARCHAR(50) NOT NULL,
    changed_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
);

-- Create trigger function
CREATE FUNCTION audit_customer_changes()
RETURNS TRIGGER AS $$
BEGIN
    IF TG_OP = 'INSERT' THEN
        INSERT INTO customer_audit (customer_id, field_name, old_value, new_value, change_type, changed_by)
        SELECT
            NEW.customer_id,
            column_name,
            NULL,
            column_value,
            'INSERT',
            current_user
        FROM json_each_text(row_to_json(NEW));
    ELSIF TG_OP = 'UPDATE' THEN
        INSERT INTO customer_audit (customer_id, field_name, old_value, new_value, change_type, changed_by)
        SELECT
            NEW.customer_id,
            key,
            old_value,
            new_value,
            'UPDATE',
            current_user
        FROM (
            SELECT
                key,
                old_value->>key AS old_value,
                new_value->>key AS new_value
            FROM
                json_each_text(row_to_json(OLD)) old_value
            FULL OUTER JOIN
                json_each_text(row_to_json(NEW)) new_value
            USING (key)
        ) changes
        WHERE old_value IS DISTINCT FROM new_value;
    ELSIF TG_OP = 'DELETE' THEN
        INSERT INTO customer_audit (customer_id, field_name, old_value, new_value, change_type, changed_by)
        SELECT
            OLD.customer_id,
            column_name,
            column_value,
            NULL,
            'DELETE',
            current_user
        FROM json_each_text(row_to_json(OLD));
    END IF;
    RETURN NULL;
END;
$$ LANGUAGE plpgsql;

-- Attach trigger
CREATE TRIGGER customer_audit_trigger
AFTER INSERT OR UPDATE OR DELETE ON customers
FOR EACH ROW EXECUTE FUNCTION audit_customer_changes();
```
